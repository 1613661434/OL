227 template <typename T>
228 auto format_arg(T&& arg) -> decltype(auto)
229 {
230 if constexpr (std::is_same_v<std::decay_t<T>, std::string>)
231 {
232 return arg.c_str();
233 }
234 else
235 {
236 return std::forward<T>(arg);
237 }
238 }

249 template <typename... Types>
250 bool sformat(std::string& str, const char* fmt, Types&&... args)
251 {
252 // 计算长度
253 int len = std::snprintf(nullptr, 0, fmt, base::format_arg(std::forward<Types>(args))...);
254 if (len < 0) return false;
255
256 if (len == 0)
257 {
258 str.clear();
259 return true;
260 }
261
262 // 执行格式化
263 str.resize(len);
264 std::snprintf(&str[0], len + 1, fmt, base::format_arg(std::forward<Types>(args))...);
265 return true;
266 }

275 template <typename... Types>
276 std::string sformat(const char* fmt, Types&&... args)
277 {
278 std::string str;
279 int len = std::snprintf(nullptr, 0, fmt, base::format_arg(std::forward<Types>(args))...);
280 if (len <= 0) return str;
281
282 str.resize(len);
283 std::snprintf(&str[0], len + 1, fmt, base::format_arg(std::forward<Types>(args))...);
284 return str;
285 }
