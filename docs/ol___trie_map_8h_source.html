62 void traverse(std::shared_ptr<TrieNode<V>> node, std::string& path, std::list<std::string>& res)
63 {
64 if (!node) return;
65
66 if (node->isValid) // 用标记位判断是否为有效键
67 {
68 res.push_back(path);
69 }
70
71 for (auto& [c, child] : node->children)
72 {
73 path.push_back(c);
74 traverse(child, path, res);
75 path.pop_back();
76 }
77 }

87 void traverseByPattern(std::shared_ptr<TrieNode<V>> node, std::string& path,
88 const std::string& pattern, size_t i, std::list<std::string>& res)
89 {
90 if (!node) return; // 节点不存在，匹配失败
91
92 // 模式匹配完成
93 if (i == pattern.length())
94 {
95 // 如果当前节点为有效键
96 if (node->isValid)
97 {
98 res.push_back(path);
99 }
100 return;
101 }
102
103 unsigned char c = pattern[i];
104 if (c == '.')
105 {
106 // 通配符匹配任意字符，尝试所有可能的子节点
107 for (auto& [ch, child] : node->children)
108 {
109 path.push_back(ch);
110 traverseByPattern(child, path, pattern, i + 1, res);
111 path.pop_back();
112 }
113 }
114 else
115 {
116 // 普通字符，查找对应的子节点
117 auto it = node->children.find(c);
118 if (it != node->children.end())
119 {
120 path.push_back(c);
121 traverseByPattern(it->second, path, pattern, i + 1, res);
122 path.pop_back();
123 }
124 }
125 }

135 std::shared_ptr<TrieNode<V>> put(std::shared_ptr<TrieNode<V>> node, const std::string& key, V val, size_t i)
136 {
137 if (!node)
138 {
139 node = std::make_shared<TrieNode<V>>();
140 }
141
142 if (i == key.length())
143 {
144 node->val = val;
145 node->isValid = true; // 标记为有效节点（无论val是否为默认值）
146 return node;
147 }
148
149 unsigned char c = key[i];
150 node->children[c] = put(node->children[c], key, val, i + 1);
151 return node;
152 }

161 std::shared_ptr<TrieNode<V>> remove(std::shared_ptr<TrieNode<V>> node, const std::string& key, size_t i)
162 {
163 if (!node) return nullptr; // 节点不存在，直接返回
164
165 // 到达键的末尾，标记为无效
166 if (i == key.length())
167 {
168 node->isValid = false;
169 }
170 else
171 {
172 unsigned char c = key[i];
173 // 递归删除子节点
174 node->children[c] = remove(node->children[c], key, i + 1);
175 }
176
177 // 后序处理：如果节点有效（isValid=true），保留节点
178 if (node->isValid)
179 {
180 return node;
181 }
182
183 // 如果没有子节点，删除当前节点
184 if (node->children.empty())
185 {
186 return nullptr;
187 }
188
189 // 有子节点但无值，保留节点作为中间路径
190 return node;
191 }

200 bool hasPattern(std::shared_ptr<TrieNode<V>> node, const std::string& pattern, size_t i)
201 {
202 if (!node) return false; // 节点不存在，匹配失败
203
204 // 模式处理完毕，检查当前节点是否有值
205 if (i == pattern.length())
206 {
207 return node->isValid;
208 }
209
210 unsigned char c = pattern[i];
211 if (c != '.')
212 {
213 // 普通字符，查找对应子节点并递归匹配
214 auto it = node->children.find(c);
215 return (it != node->children.end()) && hasPattern(it->second, pattern, i + 1);
216 }
217 else
218 {
219 // 通配符，尝试所有子节点
220 for (auto& [ch, child] : node->children)
221 {
222 if (hasPattern(child, pattern, i + 1))
223 {
224 return true;
225 }
226 }
227 return false;
228 }
229 }

237 std::shared_ptr<TrieNode<V>> findNode(std::shared_ptr<TrieNode<V>> node, const std::string& key)
238 {
239 auto current = node;
240 for (size_t i = 0; i < key.length(); ++i)
241 {
242 if (!current) return nullptr; // 路径中断，返回null
243
244 // 查找下一个字符对应的子节点
245 auto it = current->children.find(key[i]);
246 if (it == current->children.end())
247 {
248 return nullptr; // 子节点不存在，返回null
249 }
250
251 current = it->second;
252 }
253 return current; // 返回键末尾对应的节点
254 }

334 std::string shortestPrefix(const std::string& query)
335 {
336 auto current = root;
337 for (size_t i = 0; i < query.length(); ++i)
338 {
339 if (!current) break;
340
341 // 基于isValid判断，而非val是否为默认值
342 if (current->isValid)
343 {
344 return query.substr(0, i);
345 }
346
347 auto it = current->children.find(query[i]);
348 if (it == current->children.end()) break;
349 current = it->second;
350 }
351
352 if (current && current->isValid)
353 {
354 return query;
355 }
356
357 return "";
358 }

365 std::string longestPrefix(const std::string& query)
366 {
367 auto current = root;
368 size_t max_len = 0; // 记录最长有效前缀的长度
369 size_t i; // 循环变量（在循环外定义，用于判断是否遍历完所有字符）
370
371 for (i = 0; i < query.length(); ++i)
372 {
373 if (!current) break; // 路径中断，退出循环
374
375 // 如果当前节点是有效键，更新最长前缀长度
376 if (current->isValid)
377 {
378 max_len = i;
379 }
380
381 // 查找下一个字符对应的子节点
382 auto it = current->children.find(query[i]);
383 if (it == current->children.end()) break; // 字符不匹配，退出循环
384 current = it->second; // 移动到子节点
385 }
386
387 // 只有当：1. 遍历完查询字符串的所有字符；2. 最终节点是有效键
388 // 才返回整个查询字符串（说明查询字符串本身就是已插入的键）
389 if (i == query.length() && current && current->isValid)
390 {
391 return query;
392 }
393
394 // 否则返回最长有效前缀
395 return query.substr(0, max_len);
396 }
