101 template <typename... Args>
102 void addEdge(NodeType from, NodeType to, Args&&... args)
103 {
104 addNode(from); // 确保起点存在
105 addNode(to); // 确保终点存在
106
107 // 有权图需要传入权重参数，无权图不需要
108 if constexpr (IsWeighted)
109 {
110 adjList[from].emplace_back(to, std::forward<Args>(args)...);
111 }
112 else
113 {
114 adjList[from].emplace_back(to);
115 }
116
117 // 无向图自动添加反向边
118 if constexpr (!IsDirected)
119 {
120 if constexpr (IsWeighted)
121 {
122 adjList[to].emplace_back(from, std::forward<Args>(args)...);
123 }
124 else
125 {
126 adjList[to].emplace_back(from);
127 }
128 }

137 void rmEdge(NodeType from, NodeType to)
138 {
139 // 先处理正向边
140 if (adjList.find(from) != adjList.end())
141 {
142 auto& edges = adjList[from];
143 for (auto it = edges.begin(); it != edges.end(); ++it)
144 {
145 if (it->to == to)
146 {
147 edges.erase(it);
148 break;
149 }
150 }
151 }
152
153 // 无向图需要删除反向边
154 if constexpr (!IsDirected)
155 {
156 if (adjList.find(to) != adjList.end())
157 {
158 auto& reverseEdges = adjList[to];
159 for (auto it = reverseEdges.begin(); it != reverseEdges.end(); ++it)
160 {
161 if (it->to == from)
162 {
163 reverseEdges.erase(it);
164 break;
165 }
166 }
167 }
168 }
169 }

177 bool hasEdge(NodeType from, NodeType to) const
178 {
179 if (adjList.find(from) == adjList.end()) return false;
180
181 const auto& edges = adjList.at(from);
182 for (const auto& edge : edges)

184 if (edge.to == to) return true;

196 template <bool W = IsWeighted, typename = std::enable_if_t<W>>
197 WeightType weight(NodeType from, NodeType to) const
198 {
199 if (adjList.find(from) == adjList.end())
200 {
201 throw std::invalid_argument("Node does not exist: from");
202 }
203
204 const auto& edges = adjList.at(from);
205 for (const auto& edge : edges)
206 {
207 if (edge.to == to) return edge.weight;
208 }
209 throw std::invalid_argument("Edge does not exist");
210 }

218 const std::vector<Edge<NodeType, IsWeighted, WeightType>>& neighbors(NodeType node) const
219 {

221 auto it = adjList.find(node);
222 if (it == adjList.end())
223 {
224 return emptyEdges; // 返回静态空vector
225 }
226 return it->second;
227 }
