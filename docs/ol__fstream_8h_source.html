390 template <typename... Types>
391 bool write(const char* fmt, Types... args)
392 {
393 if (fout.is_open() == false) return false;
394
395 backup(); // 判断是否需要切换日志文件。
396
397 m_splock.lock(); // 加锁。
398 fout << ltime1() << " " << sformat(fmt, args...); // 把当前时间和日志内容写入日志文件。
399 m_splock.unlock(); // 解锁。
400
401 return fout.good();
402 }
403
411 template <typename T>
412 clogfile& operator<<(const T& value)
413 {

415 fout << value;
416 m_splock.unlock();
417
418 return *this;
419 }
