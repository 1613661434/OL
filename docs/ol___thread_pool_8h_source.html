2/*
3 * 程序名：ol_ThreadPool.h
4 * 功能描述：通用线程池模板类的实现，支持以下特性：
5 * - 双模式支持：固定线程数模式（默认）和动态扩缩容模式（通过模板参数控制）
6 * - 任务管理：支持无返回值任务（addTask）和带返回值任务（submitTask）
7 * - 队列策略：任务队列满时可选择拒绝、阻塞等待或超时等待策略
8 * - 线程安全：通过互斥锁和条件变量保证多线程环境下的操作安全性
9 * - 动态特性（当模板参数IsDynamic=true时）：
10 * - 自动根据任务负载扩缩容线程数量（在minThreads和maxThreads范围内）
11 * - 可配置管理者线程检查间隔，平衡响应速度和资源消耗
12 * 作者：ol
13 * 适用标准：C++17及以上（需支持constexpr if、模板条件类型等特性）

97 template <bool D = IsDynamic, typename = std::enable_if_t<!D>>
98 ThreadPool(size_t threadNum, size_t maxQueueSize = 0)
99 : m_stop(false), m_activeWorkers(0),
100 m_maxQueueSize(maxQueueSize),
101 m_queueFullPolicy(QueueFullPolicy::kReject), m_timeoutMS(std::chrono::milliseconds(500))
102 {
103 if (threadNum == 0)
104 {
105 m_stop = true;
106 return;
107 }
108
109 // 启动固定数量的工作线程
110 m_workers.reserve(threadNum);
111 while (threadNum > 0)
112 {
113 m_activeWorkers.fetch_add(1, std::memory_order_acq_rel);
114 m_workers.emplace_back(&ThreadPool<IsDynamic>::worker, this);
115 --threadNum;
116 }

128 template <bool D = IsDynamic, typename = std::enable_if_t<D>>
129 ThreadPool(size_t minThreadNum = 0,
130 size_t maxThreadNum = std::thread::hardware_concurrency(),
131 size_t maxQueueSize = 0,
132 std::chrono::seconds checkInterval = std::chrono::seconds(1))
133 : m_stop(false), m_activeWorkers(0),
134 m_maxQueueSize(maxQueueSize),
135 m_queueFullPolicy(QueueFullPolicy::kReject), m_timeoutMS(std::chrono::milliseconds(500))
136 {
137 if (minThreadNum > maxThreadNum)
138 throw std::invalid_argument("[ol::ThreadPool] Invalid thread number range");
139
140 if (minThreadNum == maxThreadNum && minThreadNum == 0)
141 {
142 m_stop = true;
143 return;
144 }
145
146 // 初始化动态模式成员
147 m_dynamic.minThreads = minThreadNum;
148 m_dynamic.maxThreads = maxThreadNum;
149 m_dynamic.idleThreads = 0;
150 m_dynamic.workerExitNum = 0;
151 m_dynamic.checkInterval = checkInterval;
152
153 // 启动最小数量（至少为1）的工作线程
154 size_t needThreads = minThreadNum == 0 ? 1 : minThreadNum;
155
156 while (needThreads > 0)
157 {
158 m_activeWorkers.fetch_add(1, std::memory_order_acq_rel);
159 std::thread th(&ThreadPool<IsDynamic>::worker, this);
160#ifdef DEBUG
161 printf("构造函数：新工作线程ID：%zu\n", th.get_id());
162#endif
163 m_workers.emplace(th.get_id(), std::move(th)); // 移动到哈希表
164 --needThreads;
165 }
166
167 // 启动管理者线程
168 m_dynamic.managerThread = std::thread(&ThreadPool<IsDynamic>::manager<IsDynamic>, this);
169#ifdef DEBUG
170 printf("构造函数：新管理者线程ID：%zu\n", m_dynamic.managerThread.get_id());
171#endif
172 }

184 int wait_ms = 0;
185 while (m_activeWorkers.load(std::memory_order_acquire) > 0 && wait_ms < 1000)

194 }
195#endif

204 void stop()
205 {
206#ifdef DEBUG
207 printf("线程池开始stop()\n");
208#endif
209
210 // 原子交换，确保仅执行一次stop逻辑 + 内存可见性
211 bool expected = false;
212 if (!m_stop.compare_exchange_strong(expected, true, std::memory_order_release, std::memory_order_relaxed))
213 {
214#ifdef DEBUG
215 printf("线程池已停止，无需重复操作\n");
216#endif
217 return;
218 }
219
220 // 确保 m_stop 对所有线程可见
221 std::atomic_thread_fence(std::memory_order_seq_cst);
222
223 // 动态模式：先停止管理者线程（确保其不再修改m_workers）
224 if constexpr (IsDynamic)
225 {
226 // 唤醒管理者线程，使其退出循环
227 m_dynamic.managerExit_condVar.notify_one();
228 if (m_dynamic.managerThread.joinable())
229 {
230 try
231 {
232 auto manager_id = m_dynamic.managerThread.get_id();
233 m_dynamic.managerThread.join();
234#ifdef DEBUG
235 printf("动态模式：管理者线程（ID:%zu）已join\n", manager_id);
236#endif
237 m_dynamic.managerThread = std::thread();
238 }
239 catch (const std::system_error& e)
240 {
241 fprintf(stderr, "[ol::ThreadPool] Dynamic mode: Manager thread join failed: %s\n", e.what());
242 m_dynamic.managerThread = std::thread();
243 }
244 }
245
246 // 清空工作线程退出队列
247 std::lock_guard<std::mutex> lock_exit_deque(m_dynamic.workerExitId_dequeMutex);
248 m_dynamic.workerExitId_deque.clear();
249#ifdef DEBUG
250 printf("动态模式：清空工作线程退出队列\n");
251#endif
252 }
253
254 // 唤醒所有等待的工作线程
255 m_taskQueueNotEmpty_condVar.notify_all();
256 m_taskQueueNotFull_condVar.notify_all();
257
258 // 等待活跃线程退出（最多等待5秒）
259 int wait_ms = 0;
260 while (m_activeWorkers.load(std::memory_order_acquire) > 0 && wait_ms < 5000)
261 {
262 std::this_thread::sleep_for(std::chrono::milliseconds(10));
263 wait_ms += 10;
264 }
265#ifdef DEBUG
266 if (m_activeWorkers.load() > 0)
267 {
268 printf("警告：仍有%d个活跃线程未退出（已等待%dms）\n",
269 (int)m_activeWorkers.load(), wait_ms);
270 }
271#endif
272
273 // 处理工作线程
274 std::lock_guard<std::mutex> lock(m_workersMutex);
275 if constexpr (IsDynamic)
276 {
277 // 动态模式：哈希表遍历
278 for (auto& [id, th] : m_workers)
279 {
280 if (th.joinable())
281 {
282#ifdef DEBUG
283 printf("动态模式：处理工作线程ID：%zu（stop）\n", id);
284#endif
285 try
286 {
287 th.join();
288 }
289 catch (const std::system_error& e)
290 {
291 fprintf(stderr, "[ol::ThreadPool] Dynamic mode: Thread (ID: %zu) join failed: %s\n", id, e.what());
292 }
293 }
294#ifdef DEBUG
295 else
296 {
297 printf("动态模式：线程(ID:%zu)不可join，跳过\n", id);
298 }
299#endif
300 }
301 }
302 else
303 {
304 // 固定模式：向量遍历
305 for (auto& th : m_workers)
306 {
307 if (th.joinable())
308 {
309#ifdef DEBUG
310 printf("固定模式：处理工作线程ID：%zu（stop）\n", th.get_id());
311#endif
312 try
313 {
314 th.join();
315 }
316 catch (const std::system_error& e)
317 {
318 fprintf(stderr, "[ol::ThreadPool] Fixed mode: Thread join failed: %s\n", e.what());
319 }
320 }
321#ifdef DEBUG
322 else
323 {
324
325 printf("固定模式：线程不可join，跳过\n");
326 }
327#endif
328 }
329 }
330
331 // 清理线程容器
332 m_workers.clear();
333#ifdef DEBUG
334 printf("线程池finish stop()\n");
335#endif
336 }

424 bool addTask(std::function<void()> task)
425 {
426 if (m_stop.load(std::memory_order_acquire)) return false;
427
428 {
429 std::unique_lock<std::mutex> lock(m_taskQueueMutex);
430
431 // 处理队列大小限制
432 if (m_maxQueueSize > 0)
433 {
434 while (m_taskQueue.size() >= m_maxQueueSize && !m_stop.load(std::memory_order_relaxed))
435 {
436 switch (m_queueFullPolicy)
437 {
438 case QueueFullPolicy::kReject:
439 return false;
440 case QueueFullPolicy::kBlock:
441 m_taskQueueNotFull_condVar.wait(lock, [this]()
442 { return m_taskQueue.size() < m_maxQueueSize || m_stop.load(std::memory_order_relaxed); });
443 break;
444 case QueueFullPolicy::kTimeout:
445 bool result = m_taskQueueNotFull_condVar.wait_for(lock, m_timeoutMS,
446 [this]()
447 { return m_taskQueue.size() < m_maxQueueSize || m_stop.load(std::memory_order_relaxed); });
448 if (!result) return false;
449 break;
450 }
451 }
452 }
453
454 if (m_stop.load(std::memory_order_relaxed)) return false;
455 m_taskQueue.push(std::move(task));
456 }
457
458 m_taskQueueNotEmpty_condVar.notify_one();
459 return true;
460 }

473 template <typename F, typename... Args>
474 auto submitTask(F&& f, Args&&... args) -> std::pair<bool, std::future<typename std::invoke_result_t<F, Args...>>>
475 {
476 using ReturnType = typename std::invoke_result_t<F, Args...>;
477
478 auto task = std::make_shared<std::packaged_task<ReturnType()>>(
479 [f = std::forward<F>(f), args = std::make_tuple(std::forward<Args>(args)...)]() mutable
480 {
481 return std::apply(std::move(f), std::move(args));
482 });
483
484 std::future<ReturnType> result = task->get_future();
485
486 if (!addTask([task]() mutable
487 { (*task)(); }))
488 {
489 std::promise<ReturnType> promise;
490 if (m_stop.load(std::memory_order_acquire))
491 {
492 promise.set_exception(std::make_exception_ptr(std::runtime_error("[ol::ThreadPool] ThreadPool has been stopped")));
493 return {false, promise.get_future()};
494 }
495
496 switch (m_queueFullPolicy)
497 {
498 case QueueFullPolicy::kReject:
499 promise.set_exception(std::make_exception_ptr(std::runtime_error("[ol::ThreadPool] Task queue full (Reject policy)")));
500 return {false, promise.get_future()};
501 case QueueFullPolicy::kBlock:
502 // 此时失败一定是因为线程池已停止（否则wait会一直等）
503 promise.set_exception(std::make_exception_ptr(std::runtime_error("[ol::ThreadPool] Task submission failed in block policy (ThreadPool stopped)")));
504 return {false, promise.get_future()};
505 case QueueFullPolicy::kTimeout:
506 promise.set_exception(std::make_exception_ptr(std::runtime_error("[ol::ThreadPool] Task queue full (Timeout policy)")));
507 return {false, promise.get_future()};
508 default:
509 promise.set_exception(std::make_exception_ptr(std::runtime_error("[ol::ThreadPool] Task submission failed")));
510 return {false, promise.get_future()};
511 }
512 }
513
514 return {true, std::move(result)};
515 }

530 void worker()
531 {
532 // 动态模式：初始化线程状态
533 if constexpr (IsDynamic)
534 {
535 m_dynamic.idleThreads.fetch_add(1, std::memory_order_release);
536 }
537
538 try
539 {
540 while (!m_stop.load(std::memory_order_acquire))
541 {
542 std::function<void()> task;
543
544 {
545 std::unique_lock<std::mutex> lock(m_taskQueueMutex);
546
547 // 等待任务或停止信号
548 auto waitCond = [this]()
549 {
550 if constexpr (IsDynamic)
551 return !m_taskQueue.empty() || m_stop.load(std::memory_order_relaxed) || m_dynamic.workerExitNum.load(std::memory_order_relaxed) > 0;
552 else
553 return !m_taskQueue.empty() || m_stop.load(std::memory_order_relaxed);
554 };
555 m_taskQueueNotEmpty_condVar.wait(lock, waitCond);
556
557 if (m_stop.load(std::memory_order_acquire)) break;
558
559 if constexpr (IsDynamic)
560 {
561 if (m_dynamic.workerExitNum.load(std::memory_order_acquire) > 0)
562 {
563 m_dynamic.workerExitNum.fetch_sub(1, std::memory_order_acq_rel);
564 break;
565 }
566 }
567
568 if (m_taskQueue.empty())
569 {
570 lock.unlock();
571 continue;
572 }
573
574 // 取出任务
575 task = std::move(m_taskQueue.front());
576 m_taskQueue.pop();
577
578 // 通知可能等待的生产者
579 if (m_queueFullPolicy != QueueFullPolicy::kReject) m_taskQueueNotFull_condVar.notify_one();
580
581 // 动态模式：空闲线程数-1
582 if constexpr (IsDynamic) m_dynamic.idleThreads.fetch_sub(1, std::memory_order_acq_rel);
583 }
584
585 // 执行任务
586 try
587 {
588 if (task) task(); // 空任务保护
589 }
590 catch (const std::exception& e)
591 {
592 fprintf(stderr, "[ol::ThreadPool] Task error: %s\n", e.what());
593 }
594 catch (...)
595 {
596 fprintf(stderr, "[ol::ThreadPool] Unknown task error\n");
597 }
598
599 // 动态模式：任务完成，恢复空闲状态
600 if constexpr (IsDynamic)
601 {
602 m_dynamic.idleThreads.fetch_add(1, std::memory_order_release);
603 }
604 }
605 }
606 catch (...)
607 {
608 fprintf(stderr, "[ol::ThreadPool] Worker thread(ID:%zu) unexpected exception\n", std::this_thread::get_id());
609 }
610
611 // 活跃线程数-1
612 m_activeWorkers.fetch_sub(1, std::memory_order_acq_rel);
613
614 // 动态模式：空闲线程数-1
615 if constexpr (IsDynamic)
616 {
617 m_dynamic.idleThreads.fetch_sub(1, std::memory_order_acq_rel);
618 // 线程池未停止时，让管理者清理线程
619 if (!m_stop.load(std::memory_order_acquire))
620 {
621 std::unique_lock<std::mutex> lock_exitVector(m_dynamic.workerExitId_dequeMutex);
622#ifdef DEBUG
623 printf("线程(ID:%zu)加入退出容器\n", std::this_thread::get_id());
624#endif
625 m_dynamic.workerExitId_deque.emplace_back(std::this_thread::get_id());
626 }
627#ifdef DEBUG
628 else
629 {
630
631 printf("线程(ID:%zu)：线程池已停止，跳过加入退出容器\n", std::this_thread::get_id());
632 }
633#endif
634 }
635
636#ifdef DEBUG
637 printf("线程已销毁（主动移除，ID:%zu）\n", std::this_thread::get_id());
638#endif
639 }

649 template <bool D = IsDynamic, typename = std::enable_if_t<D>>
650 void manager()
651 {
652#ifdef DEBUG
653 printf("管理者线程(ID:%zu)启动\n", std::this_thread::get_id());
654#endif
655
656 try
657 {
658 std::deque<std::thread::id> exitIds;
659 while (!m_stop.load(std::memory_order_acquire))
660 {
661 // 定期检查（可被stop()唤醒）
662 std::unique_lock<std::mutex> lock_manger(m_dynamic.managerMutex);
663 m_dynamic.managerExit_condVar.wait_for(lock_manger, m_dynamic.checkInterval, [this]()
664 { return m_stop.load(std::memory_order_acquire); });
665 if (m_stop.load(std::memory_order_acquire)) return;
666
667 // 1. 清理已终止的线程对象
668 {
669 // 上锁
670 std::lock_guard<std::mutex> lock_workers(m_workersMutex);
671 std::unique_lock<std::mutex> lock_exitDeque(m_dynamic.workerExitId_dequeMutex);
672
673 // 交换退出线程ID队列
674 exitIds.clear();
675 exitIds.swap(m_dynamic.workerExitId_deque);
676 lock_exitDeque.unlock(); // 解锁
677
678 // 清理退出的线程
679 for (const auto& exitId : exitIds)
680 {
681#ifdef DEBUG
682 printf("待清理线程ID：%zu\n", exitId);
683#endif
684 auto it = m_workers.find(exitId);
685 if (it == m_workers.end())
686 {
687#ifdef DEBUG
688 printf("线程ID已被清理，跳过: %zu\n", exitId);
689#endif
690 continue;
691 }
692
693 if (it->second.joinable())
694 {
695 try
696 {
697 it->second.join();
698#ifdef DEBUG
699 printf("线程ID已join: %zu\n", exitId);
700#endif
701 }
702 catch (const std::exception& e)
703 {
704 fprintf(stderr, "[ol::ThreadPool] Worker thread join failure: %s\n", e.what());
705 }
706 catch (...)
707 {
708 fprintf(stderr, "[ol::ThreadPool] Unknown Worker thread join error\n");
709 }
710 }
711
712 m_workers.erase(it);
713 }
714#ifdef DEBUG
715 printf("线程从容器移除（管理者清理）\n");
716#endif
717 }
718
719 // 2. 扩缩容
720 {
721 std::lock_guard<std::mutex> lock_taskQueue(m_taskQueueMutex);
722 std::lock_guard<std::mutex> lock_workers(m_workersMutex);
723 size_t taskCount = m_taskQueue.size();
724 size_t workerCount = m_workers.size();
725 size_t idleCount = m_dynamic.idleThreads.load(std::memory_order_acquire);
726
727 // 扩容判断：任务数 > 线程数 * 2 且未达最大线程数
728 if (taskCount > workerCount * 2 && workerCount < m_dynamic.maxThreads)
729 {
730 size_t needThreads = std::min(
731 m_dynamic.maxThreads - workerCount,
732 (taskCount + workerCount - 1) / workerCount // 按当前负载估算需要的线程数（向上取整）
733 );
734 // 每次最多扩容到当前的1.5倍，避免一次性创建过多线程
735 needThreads = std::min(needThreads, workerCount / 2 + 1);
736
737 while (needThreads > 0)
738 {
739 m_activeWorkers.fetch_add(1, std::memory_order_acq_rel);
740 std::thread th(&ThreadPool<IsDynamic>::worker, this);
741#ifdef DEBUG
742 printf("新线程（ID: %zu）\n", th.get_id());
743#endif
744 m_workers.emplace(th.get_id(), std::move(th)); // 哈希表插入新线程
745 --needThreads;
746 }
747#ifdef DEBUG
748 printf("扩容：线程数从 %zu 增加到 %zu（任务数: %zu）\n",
749 workerCount, m_workers.size(), taskCount);
750#endif
751 }
752 // 缩容判断：空闲线程 > 线程总数的1/2 且 线程数超过「最小线程数或1（取较大值）」
753 else if (idleCount > workerCount / 2 && workerCount > std::max(m_dynamic.minThreads, static_cast<size_t>(1)))
754 {
755 // 缩容下限：最多缩减到「最小线程数或1（取较大值）」
756 size_t minKeep = std::max(m_dynamic.minThreads, static_cast<size_t>(1));
757
758 // 实际缩减数 = 取「可缩减线程数」和「多余空闲线程数」的较小值
759 size_t reduceThreads = std::min(
760 workerCount - minKeep, // 可缩减线程数 = 当前线程数 - 最低保留数
761 idleCount - (workerCount / 2) // 只销毁超过一半的空闲线程
762 );
763
764#ifdef DEBUG
765 size_t reduceThreads_temp = reduceThreads;
766#endif
767
768 // 销毁线程
769 if (reduceThreads > 0)
770 {
771 m_dynamic.workerExitNum.store(reduceThreads, std::memory_order_release);
772 do
773 {
774 m_taskQueueNotEmpty_condVar.notify_one();
775 --reduceThreads;
776 } while (reduceThreads > 0);
777 }
778#ifdef DEBUG
779 printf("缩容：计划销毁 %zu 个线程（当前线程数: %zu, 空闲数: %zu, 保留至少: %zu）\n",
780 reduceThreads_temp, workerCount, idleCount, minKeep);
781#endif
782 }
783 }
784 }
785 }
786 catch (...)
787 {
788 fprintf(stderr, "[ol::ThreadPool] Manager thread(ID:%zu) unexpected exception\n", std::this_thread::get_id());
789 }
790
791 // 清空退出队列
792 std::lock_guard<std::mutex> lock_exit_deque(m_dynamic.workerExitId_dequeMutex);
793 m_dynamic.workerExitId_deque.clear();
794#ifdef DEBUG
795 printf("管理者线程(ID:%zu)退出，清空退出队列\n", std::this_thread::get_id());
796#endif
797 }
798 };
