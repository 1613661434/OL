32 // 跨平台兼容：将localtime_s/localtime_r的结果统一为bool返回
33 inline bool localtime_now(struct tm* tm_out, const time_t* t)
34 {
35 // 基本参数校验
36 if (tm_out == nullptr || t == nullptr)
37 {
38 return false; // 无效指针直接返回失败
39 }
40
41#ifdef _WIN32
42 // Windows：localtime_s返回errno_t，0表示成功
43 errno_t err = localtime_s(tm_out, t);
44 return (err == 0); // 成功返回true，失败返回false
45#else
46 // Linux/Unix/macOS：localtime_r返回struct tm*，非NULL表示成功
47 struct tm* ret = localtime_r(t, tm_out);
48 return (ret != nullptr); // 成功返回true，失败返回false
49#endif
50 }

54 /*
55 取操作系统的时间（用字符串表示）。
56 strtime：用于存放获取到的时间。
57 timetvl：时间的偏移量，单位：秒，0是缺省值，表示当前时间，30表示当前时间30秒之后的时间点，-30表示当前时间30秒之前的时间点。
58 fmt：输出时间的格式，fmt每部分的含义：yyyy-年份；mm-月份；dd-日期；hh24-小时；mi-分钟；ss-秒，
59 缺省是"yyyy-mm-dd hh24:mi:ss"，目前支持以下格式：
60 "yyyy-mm-dd hh24:mi:ss"
61 "yyyymmddhh24miss"
62 "yyyy-mm-dd"
63 "yyyymmdd"
64 "hh24:mi:ss"
65 "hh24miss"
66 "hh24:mi"
67 "hh24mi"
68 "hh24"
69 "mi"
70 注意：
71 1）小时的表示方法是hh24，不是hh，这么做的目的是为了保持与数据库的时间表示方法一致；
72 2）以上列出了常用的时间格式，如果不能满足你应用开发的需求，请修改源代码timetostr()函数增加更多的格式支持；
73 3）调用函数的时候，如果fmt与上述格式都匹配，strtime的内容将为空。
74 4）时间的年份是四位，其它的可能是一位和两位，如果不足两位，在前面补0。
