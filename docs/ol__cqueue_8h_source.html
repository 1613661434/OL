2/*
3 * 程序名：ol_cqueue.h
4 * 功能描述：固定大小的循环队列（circular queue）模板类，支持以下特性：
5 * - 基于静态数组实现，大小由模板参数指定
6 * - 支持POD类型和非POD类型，自动适配内存管理
7 * - 提供入队、出队、查看队头、判断空满等基础操作
8 * - 支持移动构造和移动赋值，禁用拷贝构造和赋值（避免资源冲突）
9 * - 支持原地构造元素（emplace）以提升性能
10 * 作者：ol
11 * 适用标准：C++11及以上（需支持constexpr、type_traits、右值引用等特性）
12 */

33 template <class T, size_t MAX_SIZE>
34 class cqueue
35 {
36 private:
37 static_assert(MAX_SIZE > 0, "MAX_SIZE must be greater than 0");
38 bool m_inited = false; // 队列被初始化标志，true-已初始化；false-未初始化。
39 size_t m_size = 0; // 队列的实际长度。
40 T m_data[MAX_SIZE]; // 用数组存储循环队列中的元素。
41 size_t m_front = 0; // 队列的头指针。
42 size_t m_rear = MAX_SIZE - 1; // 队列的尾指针，指向队尾元素。
43
44 cqueue(const cqueue&) = delete; // 禁用拷贝构造函数。
45 cqueue& operator=(const cqueue&) = delete; // 禁用赋值函数。
46
47 public:
48 // 构造函数，自动初始化队列
49 cqueue()
50 {
51 init();
52 }
53
54 // 析构函数，释放资源（非平凡析构类型需手动调用析构函数）
55 ~cqueue()
56 {
57 if (m_inited == true)
58 {
59 // 非可平凡复制类型需要手动调用析构函数
60 if constexpr (!std::is_trivially_destructible_v<T>)
61 {
62 for (size_t i = 0; i < m_size; ++i)
63 {
64 size_t index = (m_front + i) % MAX_SIZE;
65 m_data[index].~T(); // 显式调用析构函数
66 }
67 }
68 }
69 }

75 cqueue(cqueue&& other) noexcept
76 : m_inited(other.m_inited),
77 m_size(other.m_size),
78 m_front(other.m_front),
79 m_rear(other.m_rear)
80 {
81 if (m_inited == true)
82 {
83 // 移动可平凡复制类型：直接内存拷贝
84 if constexpr (std::is_trivially_copyable_v<T>)
85 {
86 memcpy(m_data, other.m_data, MAX_SIZE * sizeof(T));
87 }
88 else
89 {
90 // 非可平凡复制类型：逐个移动
91 for (size_t i = 0; i < other.m_size; ++i)
92 {
93 size_t src_idx = (other.m_front + i) % MAX_SIZE;
94 size_t dst_idx = (m_front + i) % MAX_SIZE;
95 m_data[dst_idx] = std::move(other.m_data[src_idx]);
96 }
97 }
98 // 清空原队列
99 other.m_inited = false;
100 other.m_size = 0;
101 }
102 }

109 cqueue& operator=(cqueue&& other) noexcept
110 {
111 if (this != &other)
112 {
113 // 先释放当前队列的资源
114 if (m_inited == true)
115 {
116 if constexpr (!std::is_trivially_destructible_v<T>)
117 {
118 // 非可平凡析构类型：手动调用析构函数
119 for (size_t i = 0; i < m_size; ++i)
120 {
121 size_t index = (m_front + i) % MAX_SIZE;
122 m_data[index].~T(); // 显式调用析构函数
123 }
124 }
125 // 无论是否可平凡析构，都需要重置状态
126 m_inited = false;
127 m_size = 0;
128 m_front = 0;
129 m_rear = MAX_SIZE - 1;
130 }
131
132 // 移动其他队列的资源
133 m_inited = other.m_inited;
134 m_size = other.m_size;
135 m_front = other.m_front;
136 m_rear = other.m_rear;
137
138 if (m_inited == true)
139 {
140 if constexpr (std::is_trivially_copyable_v<T>)
141 {
142 // 可平凡复制类型：直接内存拷贝
143 memcpy(m_data, other.m_data, MAX_SIZE * sizeof(T));
144 }
145 else
146 {
147 // 非可平凡复制类型：逐个移动元素
148 for (size_t i = 0; i < other.m_size; ++i)
149 {
150 size_t src_idx = (other.m_front + i) % MAX_SIZE;
151 size_t dst_idx = (m_front + i) % MAX_SIZE;
152 m_data[dst_idx] = std::move(other.m_data[src_idx]);
153 }
154 }
155 // 清空原队列
156 other.m_inited = false;
157 other.m_size = 0;
158 }
159 }
160 return *this;
161 }

168 void init()
169 {
170 if (m_inited == true) return; // 循环队列的初始化只能执行一次。
171 m_inited = true;
172 m_front = 0; // 头指针。
173 m_rear = MAX_SIZE - 1; // 为了方便写代码，初始化时，尾指针指向队列的最后一个位置。
174 m_size = 0; // 队列的实际长度。
175
176 // 数组元素初始化。
177 if constexpr (std::is_trivially_copyable_v<T>)
178 {
179 // 可平凡复制类型专用初始化（效率优先）
180 memset(m_data, 0, sizeof(m_data));
181 }
182 else
183 {
184 for (size_t i = 0; i < MAX_SIZE; ++i)
185 {
186 // 非可平凡复制类型通用初始化
187 m_data[i] = T(); // 调用默认构造函数
188 }
189 }
190 }

253 bool pop()
254 {
255 if (empty()) return false;
256
257 m_front = (m_front + 1) % MAX_SIZE; // 队列头指针后移。

267 void clear()
268 {
269 if (!m_inited || empty()) return; // 未初始化或已空则直接返回
270
271 // 处理非平凡析构类型：需要显式调用每个元素的析构函数
272 if constexpr (!std::is_trivially_destructible_v<T>)
273 {
274 for (size_t i = 0; i < m_size; ++i)
275 {
276 size_t index = (m_front + i) % MAX_SIZE;
277 m_data[index].~T(); // 显式析构元素
278 }
279 }
280
281 // 重置队列状态（无需清空数组内存，后续操作会覆盖）
282 m_front = 0;
283 m_rear = MAX_SIZE - 1;
284 m_size = 0;
285 }

324 template <typename... Args>
325 bool emplace(Args&&... args)
326 {
327 if (full())
328 {
329 std::cerr << "Circular queue is full, enqueue failed.\n";
330 return false;
331 }
332 m_rear = (m_rear + 1) % MAX_SIZE;
