43 template <typename T>
44 class UnionFind<T, std::enable_if_t<std::is_integral_v<T>>>
45 {
46 private:
47 std::vector<T> parent;
48 std::vector<size_t> rank;
49
50 public:
55 explicit UnionFind(size_t size = 0)
56 {
57 init(size);
58 }
59
64 void init(size_t size)
65 {
66 parent.resize(size);
67 rank.resize(size, 0);
68 for (size_t i = 0; i < size; ++i)
69 {
70 parent[i] = static_cast<T>(i);
71 }
72 }
73
81 T find(T x)
82 {
83 if (parent[x] != x)
84 {
85 parent[x] = find(parent[x]); // 递归压缩路径
86 }
87 return parent[x];
88 }
89
97 void unite(T x, T y)
98 {
99 T rootX = find(x);
100 T rootY = find(y);
101
102 if (rootX == rootY) return; // 已在同一集合
103
104 if (rank[rootX] < rank[rootY])
105 {
106 parent[rootX] = rootY;
107 }
108 else if (rank[rootX] > rank[rootY])
109 {
110 parent[rootY] = rootX;
111 }
112 else
113 {
114 parent[rootY] = rootX;
115 ++rank[rootX]; // 秩相同，合并后根节点秩+1
116 }
117 }
118
125 bool connected(T x, T y)
126 {
127 return find(x) == find(y);
128 }
129
134 size_t countSets() const
135 {
136 size_t count = 0;
137 for (size_t i = 0; i < parent.size(); ++i)
138 {
139 if (parent[i] == static_cast<T>(i))
140 { // 根节点的父节点是自身
141 ++count;
142 }
143 }
144 return count;
145 }
146
151 size_t size() const
152 {
153 return parent.size();
154 }
155 };

163 template <typename T>
164 class UnionFind<T, std::enable_if_t<!std::is_integral_v<T>>>
165 {
166 private:
170 struct Node
171 {
172 T parent;
173 size_t rank;
179 Node(const T& p = T(), size_t r = 0) : parent(p), rank(r)
180 {
181 }
182 };
183
184 std::unordered_map<T, Node> nodes;
185
186 public:
191 void insert(const T& x)
192 {
193 if (nodes.find(x) == nodes.end())
194 {
195 nodes[x] = Node(x); // 新元素初始父节点为自身
196 }
197 }
198
207 T find(const T& x)
208 {
209 auto it = nodes.find(x);
210 if (it == nodes.end())
211 {
212 insert(x); // 自动插入不存在的元素
213 return x;
214 }
215
216 if (it->second.parent != x)
217 {
218 it->second.parent = find(it->second.parent); // 递归压缩路径
219 }
220 return it->second.parent;
221 }
222
230 void unite(const T& x, const T& y)
231 {
232 insert(x);
233 insert(y);
234
235 T rootX = find(x);
236 T rootY = find(y);
237
238 if (rootX == rootY) return; // 已在同一集合
239
240 auto& nodeX = nodes[rootX];
241 auto& nodeY = nodes[rootY];
242
243 if (nodeX.rank < nodeY.rank)
244 {
245 nodeX.parent = rootY;
246 }
247 else if (nodeX.rank > nodeY.rank)
248 {
249 nodeY.parent = rootX;
250 }
251 else
252 {
253 nodeY.parent = rootX;
254 ++nodeX.rank; // 秩相同，合并后根节点秩+1
255 }
256 }
257
264 bool connected(const T& x, const T& y)
265 {
266 return find(x) == find(y);
267 }
268
273 size_t countSets() const
274 {
275 size_t count = 0;
276 for (const auto& pair : nodes)
277 {
278 if (pair.first == pair.second.parent)
279 { // 根节点的父节点是自身
280 ++count;
281 }
282 }
283 return count;
284 }
285
290 size_t size() const
291 {
292 return nodes.size();
293 }
294 };
