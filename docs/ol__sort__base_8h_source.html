35 template <typename Iterator, typename Compare>
36 void insertion_sort_base(Iterator first, Iterator last, const Compare& comp)
37 {
38 if (first == last) return;
39
40 // 从第二个元素开始迭代（第一个元素已"有序"）
41 for (Iterator i = std::next(first); i != last; ++i)
42 {
43 auto key = *i; // 保存当前待插入的元素
44 Iterator j = i; // 从当前位置向前查找插入点
45
46 // 向前查找：只要没到起始位置，且key应排在前一个元素前面
47 // std::prev(j)对双向迭代器等价于--j的临时值，对随机访问迭代器等价于j-1
48 while (j != first && comp(key, *std::prev(j)))
49 {
50 *j = *std::prev(j); // 前一个元素后移
51 --j; // 指针前移（双向/随机访问迭代器均支持）
52 }
53
54 *j = key; // 将key插入到正确位置
55 }
56 }

72 RandomIt binary_search_base(RandomIt first, RandomIt last,

131 template <typename RandomIt, typename Compare>
132 void shell_group_sort(RandomIt first, RandomIt last,
133 ptrdiff_t start, ptrdiff_t step, const Compare& comp)
134 {
135 const ptrdiff_t n = std::distance(first, last);
136
137 for (ptrdiff_t i = start + step; i < n; i += step)
138 {
139 auto key = *(first + i);
140 ptrdiff_t j = i - step;
141
142 while (j >= start && comp(key, *(first + j)))
143 {
144 *(first + j + step) = *(first + j);
145 j -= step;
146 }
147
148 // 这里必须要(j + step)否则VC会报错，因为VC是先first+j，导致越界直接报错
149 *(first + (j + step)) = key;
150 }
151 }

161 template <typename RandomIt, typename Compare>
162 void shell_sort_base(RandomIt first, RandomIt last, const Compare& comp)
163 {
164 const ptrdiff_t n = std::distance(first, last);
165 if (n <= 1) return;
166
167 // 使用(3^k - 1)/2的递增序列：1, 4, 13, 40, 121...
168 ptrdiff_t step = 1;
169 while (step < n / 3)
170 {
171 step = 3 * step + 1;
172 }
173
174 while (step >= 1)
175 {
176 // 对每个组执行插入排序
177 for (ptrdiff_t i = 0; i < step; ++i)
178 {
179 shell_group_sort(first, last, i, step, comp);
180 }
181 step /= 3;
182 }

196 template <typename Iterator, typename Compare>
197 void bubble_sort_base(Iterator first, Iterator last, const Compare& comp)
198 {
199 if (first == last) return;
200
201 bool swapped; // 记录是否进行过交换操作
202 Iterator end = last;
203
204 do
205 {
206 swapped = false;
207 Iterator current = first;
208 Iterator next = first;
209 ++next;
210
211 while (next != end)
212 {
213 if (comp(*next, *current))
214 {
215 std::iter_swap(current, next);
216 swapped = true;
217 }
218 ++current;
219 ++next;
220 }
221 --end; // 每轮结束后，最大元素已"冒泡"到末尾
222 } while (swapped); // 一轮内，如果一次交换操作都没有进行，说明数组已经有序，可以提前终止算法
223 }

267 template <typename RandomIt, typename Compare>
268 void heapify_base(RandomIt first, size_t size, size_t index, const Compare& comp)
269 {
270 while (true)
271 {
272 size_t largest = index;
273 size_t left = 2 * index + 1;
274 size_t right = 2 * index + 2;
275
276 if (left < size && comp(*(first + largest), *(first + left)))
277 {
278 largest = left;
279 }
280 if (right < size && comp(*(first + largest), *(first + right)))
281 {
282 largest = right;
283 }
284
285 if (largest == index)
286 {
287 break;
288 }
289
290 std::iter_swap(first + index, first + largest);
291 index = largest;
292 }

337 template <typename RandomIt, typename TempIt, typename Compare>
338 void merge_base(RandomIt first, RandomIt mid, RandomIt last,
339 TempIt temp, const Compare& comp)
340 {
341 RandomIt i = first, j = mid;
342 TempIt k = temp;
343
344 while (i < mid && j < last)
345 {
346 if (comp(*i, *j))
347 {
348 *k++ = *i++;
349 }
350 else
351 {
352 *k++ = *j++;
353 }
354 }
355
356 while (i < mid)
357 {
358 *k++ = *i++;
359 }
360
361 while (j < last)
362 {
363 *k++ = *j++;
364 }
365
366 // 将临时数组复制回原数组
367 std::copy(temp, k, first);
368 }

424 template <typename RandomIt, typename Compare>
425 typename std::enable_if<std::is_integral<typename std::iterator_traits<RandomIt>::value_type>::value, void>::type
426 counting_sort_base(RandomIt first, RandomIt last, const Compare& comp)
427 {
428 using ValueType = typename std::iterator_traits<RandomIt>::value_type;
429 const ptrdiff_t n = std::distance(first, last);
430 if (n <= 1) return;
431
432 // 找到最小值和最大值
433 ValueType min_val = *first, max_val = *first;
434 for (RandomIt it = first; it != last; ++it)
435 {
436 if (comp(*it, min_val)) min_val = *it;
437 if (comp(max_val, *it)) max_val = *it;
438 }
439
440 // 创建计数数组
441 size_t range = max_val - min_val + 1;
442 std::vector<size_t> count(range, 0);
443
444 // 计数
445 for (RandomIt it = first; it != last; ++it)
446 {
447 ++count[*it - min_val];
448 }
449
450 // 计算前缀和，得到的是在排序后数组的结束位置
451 for (size_t i = 1; i < range; ++i)
452 {
453 count[i] += count[i - 1];
454 }
455
456 // 放置元素，从后往前遍历是为了保证排序的稳定性
457 std::vector<ValueType> output(n);
458 for (RandomIt it = last - 1; it >= first; --it)
459 {
460 output[count[*it - min_val] - 1] = *it;
461 --count[*it - min_val];
462 }
463
464 // 复制回原数组
465 std::copy(output.begin(), output.end(), first);
466 }

475 template <typename T>
476 T fast_pow(T base, int exponent)
477 {
478 T result = 1;
479 while (exponent > 0)
480 {
481 // 如果当前指数为奇数，乘上当前基数
482 if (exponent % 2 == 1)
483 {
484 result *= base;
485 }
486 // 基数平方，指数折半
487 base *= base;
488 exponent /= 2;
489 }
490 return result;

500 template <typename ValueType>
501 void radix_count_lsd_sort(std::vector<ValueType>& nums, int k, int radix)
502 {
503 const size_t n = nums.size();
504 if (n <= 1) return;
505
506 // 计数数组：存储每个digit的出现次数
507 std::vector<int> count(radix, 0);
508
509 // 使用快速幂计算除数（radix^k）
510 ValueType divisor = fast_pow(static_cast<ValueType>(radix), k);
511
512 // 1. 统计当前位的数字出现次数
513 for (ValueType num : nums)
514 {
515 int digit = static_cast<int>((num / divisor) % radix);
516 ++count[digit];
517 }
518
519 // 2. 计算前缀和，确定每个数字在结果中的位置
520 for (int i = 1; i < radix; ++i)
521 {
522 count[i] += count[i - 1];
523 }
524
525 // 3. 从后往前遍历，按当前位排序（保证稳定性）
526 std::vector<ValueType> sorted(n);
527 for (int i = static_cast<int>(n) - 1; i >= 0; --i)
528 {
529 ValueType num = nums[i];
530 int digit = static_cast<int>((num / divisor) % radix);
531 sorted[count[digit] - 1] = num;
532 --count[digit];
533 }
534
535 // 4. 复制回原数组
536 nums.swap(sorted);
537 }

546 template <typename RandomIt>
547 void radix_sort_lsd_base(RandomIt first, RandomIt last, int radix)
548 {
549 using ValueType = typename std::iterator_traits<RandomIt>::value_type;
550 const ptrdiff_t n = std::distance(first, last);
551 if (n <= 1) return;
552
553 // 一次循环同时找到最大和最小元素（从第二个元素开始比较）
554 ValueType min_val = *first;
555 ValueType max_val = *first;
556 for (RandomIt it = std::next(first); it != last; ++it)
557 {
558 if (*it < min_val)
559 min_val = *it;
560 if (*it > max_val)
561 max_val = *it;
562 }
563
564 // 处理负数：计算偏移量将所有数转为非负数
565 const ValueType offset = (min_val < 0) ? -min_val : 0;
566
567 // 复制数据并应用偏移量
568 std::vector<ValueType> nums;
569 nums.reserve(n);
570 for (RandomIt it = first; it != last; ++it)
571 {
572 nums.push_back(*it + offset);
573 }
574
575 // 调整最大值（加上偏移量）
576 max_val += offset;
577
578 // 计算最大元素的位数（使用do-while确保0也能正确得到位数1）
579 int max_digits = 0;
580 ValueType temp = max_val;
581 do
582 {
583 ++max_digits;
584 temp /= radix;
585 } while (temp > 0);
586
587 // 从低位到高位，依次对每一位进行计数排序
588 for (int k = 0; k < max_digits; ++k)
589 {
590 radix_count_lsd_sort(nums, k, radix);
591 }
592
593 // 将所有元素转回原始值（减去偏移量）并写回原容器
594 auto dest_it = first;
595 for (ValueType num : nums)
596 {
597 *dest_it++ = num - offset;
598 }
599 }

624 template <typename RandomIt>
625 void radix_sort_msd_base(RandomIt first, RandomIt last,
626 size_t pos, int max_pos, int radix = 256)
627 {
628 // 递归终止条件：区间长度小于等于1，或已处理到最大指定位置
629 if (last - first <= 1 || (max_pos != -1 && pos >= static_cast<size_t>(max_pos)))
630 return;
631
632 // 创建桶：radix个桶用于存放不同字符的元素，1个额外桶用于存放短字符串
633 using ValueType = typename std::iterator_traits<RandomIt>::value_type;
634 std::vector<std::vector<ValueType>> buckets(radix + 1);
635
636 // 分配元素到对应桶（使用unsigned char避免符号扩展）
637 for (auto iter = first; iter != last; ++iter)
638 {
639 unsigned char c = get_char(*iter, pos);
640 size_t bucket_idx = static_cast<size_t>(c) + 1; // +1避开索引0
641 buckets[bucket_idx].push_back(*iter);
642 }
643
644 // 将桶中元素写回原区间，并对非空桶递归排序下一位
645 auto dest = first;
646 for (auto& bucket : buckets)
647 {
648 if (bucket.empty()) continue;
649
650 // 将当前桶元素复制回原区间
651 dest = std::copy(bucket.begin(), bucket.end(), dest);
652
653 // 递归处理下一位（空字符桶不需要继续递归）
654 if (!bucket.empty() && &bucket != &buckets[0])
655 {
656 auto bucket_first = dest - bucket.size();
657 radix_sort_msd_base(bucket_first, dest, pos + 1, max_pos, radix);
658 }
659 }

671 template <typename RandomIt>
672 std::vector<std::vector<typename std::iterator_traits<RandomIt>::value_type>>
673 radix_group_by_prefix_base(RandomIt first, RandomIt last,
674 size_t group_pos, int radix = 256)
675 {
676 // 先按指定位数进行MSD排序
677 radix_sort_msd_base(first, last, 0, static_cast<int>(group_pos), radix);
678
679 // 提取分组结果
680 using ValueType = typename std::iterator_traits<RandomIt>::value_type;
681 std::vector<std::vector<ValueType>> groups;
682 if (first == last) return groups;
683
684 // 按前缀相同性分组
685 groups.emplace_back();
686 groups.back().push_back(*first);
687 const size_t prefix_len = std::min<size_t>(group_pos, first->size());
688 std::string prev_prefix = first->substr(0, prefix_len);
689
690 for (auto iter = std::next(first); iter != last; ++iter)
691 {
692 const size_t curr_len = std::min<size_t>(group_pos, iter->size());
693 std::string curr_prefix = iter->substr(0, curr_len);
694
695 if (curr_prefix != prev_prefix)
696 {
697 groups.emplace_back();
698 prev_prefix = curr_prefix;
699 }
700 groups.back().push_back(*iter);
701 }
702

713 template <typename RandomIt, typename Compare>
714 auto median_of_three(RandomIt low, RandomIt high, const Compare& comp)
715 {
716 RandomIt mid = low + (high - low) / 2;
717
718 // 对三个位置的元素进行排序
719 if (comp(*high, *mid)) std::iter_swap(high, mid);
720 if (comp(*high, *low)) std::iter_swap(high, low);
721 if (comp(*low, *mid)) std::iter_swap(low, mid);
722
723 // 返回基准值（此时已位于low位置）
724 return *low;
725 }

731 template <typename RandomIt, typename Compare>
732 void quick_sort_base(RandomIt first, RandomIt last, const Compare& comp)
733 {
734 ptrdiff_t size = std::distance(first, last);
735
736 // 优化点：小规模数据（≤16个元素）使用插入排序
737 if (size <= 16)
738 {
739 insertion_sort_base(first, last, comp);
740 return;
741 }
742
743 // 初始化指针，high指向最后一个元素
744 RandomIt low = first;
745 RandomIt high = last - 1;
746
747 // 三数取中法选择基准值，同时调整high指针
748 auto pivot = median_of_three(low, high--, comp); // high--因为三数取中已处理最后一个元素
749
750 // 挖坑填数核心逻辑
751 while (low < high)
752 {
753 // 从右向左找小于等于基准的元素，填入左边的坑
754 while (low < high && !comp(*high, pivot)) // *high >= pivot
755 {
756 // 遇到与基准相等的元素，提前退出以平衡分区
757 if (!comp(pivot, *high)) // *high == pivot（严格弱序下：!comp(pivot, *high) && !comp(*high, pivot)）
758 {
759 --high;
760 break;
761 }
762 --high;
763 }
764 *low = *high;
765
766 // 从左向右找大于等于基准的元素，填入右边的坑
767 while (low < high && comp(*low, pivot)) // *low < pivot
768 {
769 // 遇到与基准等价的元素，提前退出以平衡分区
770 // 保留此判断以增强对不规范比较器的容错性
771 if (comp(pivot, *low)) // *low == pivot（非严格弱序下：comp(pivot, *low) && comp(*low, pivot)）
772 {
773 ++low;
774 break;
775 }
776 ++low;
777 }
778 *high = *low;
779 }
780
781 // 将基准值填入最后一个坑
782 *low = pivot;
783
784 // 递归排序左右子区间
785 quick_sort_base(first, low, comp); // 左区间：[first, low)
786 quick_sort_base(low + 1, last, comp); // 右区间：[low+1, last)
787 }
